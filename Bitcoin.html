def create_bitcoin(bitcoin, look_back=1):
    dataX, dataY = [], []
    for i in range(len(bitcoin)-look_back-1):
        a = bitcoin[i:(i+look_back), 0]
        dataX.append(a)
        dataY.append(bitcoin[i + look_back, 0])
    return np.array(dataX), np.array(dataY)
